<script>
/* --- Cesium初期設定 --- */
const viewer = new Cesium.Viewer('cesiumContainer', {
  terrainProvider: Cesium.createWorldTerrain(),
  shouldAnimate: true
});

/* --- 東京スカイツリー付近にカメラジャンプ --- */
viewer.camera.flyTo({
  destination : Cesium.Cartesian3.fromDegrees(139.8107, 35.7101, 500), // 経度, 緯度, 高さ(m)
  orientation : {
    heading : Cesium.Math.toRadians(0.0),   // 方位角（北向き）
    pitch : Cesium.Math.toRadians(-30.0),   // 下向き
    roll : 0.0
  }
});
</script>
